```sql
SELECT LPAD(' ',2*(LEVEL-1)) || tree_name, tree_lvl, tree_h_name, level, SYS_CONNECT_BY_PATH(tree_name, '-') AS PATH
FROM tree
START WITH tree_h_name is null
CONNECT BY PRIOR tree_name = tree_h_name    
ORDER SIBLINGS by tree_name
;
```
